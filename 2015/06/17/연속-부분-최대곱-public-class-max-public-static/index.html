연속 부분 최대곱

public class MAX{

  public static double[] nums = {1.1, 0.7, 1.3, 0.9, 1.4, 0.8, 0.7, 1.4};
  public static void main(String[] args){
    float max = 1;
    for(int i = 0; i < nums.length; i++){

      if(nums[i] > 1){
        float m = 1;
        for(int j = 0; j < nums.length - i; j++){

          m *= nums[i + j];

          if(max < m){
            max = m;
          }
        }
      }
    }
    System.out.println(max);
  }
}
