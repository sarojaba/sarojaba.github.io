bubble-pop

Jul 29, 2013

비누방울이 터지는 것을 표현해 보았습니다.

int min;
int max;

void setup() {
    size(320, 180);    
    frameRate(8);

    min = 0;
    max = 100;
}

void draw() {

    // clear slowly
    noStroke();
    colorMode(RGB);
    fill(color(255, 255, 255, 127));
    rect(0, 0, width, height);

    // draw circle
    colorMode(HSB);
    fill(random(255), 255, 255);  
    float diam = random(min, max);
    ellipse(random(width), random(height), diam, diam);
}
