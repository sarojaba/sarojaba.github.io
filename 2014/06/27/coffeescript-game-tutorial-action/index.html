coffeescript-game-tutorial-action

Jun 27, 2014

원이 대각선 아래로 움직이는 코드이다.

x = 150
y = 150
dx = 2
dy = 4
ctx = null

init = ->
    canvas = document.getElementById("myCanvas")
    ctx = canvas.getContext("2d")

    return setInterval(draw, 10)

draw = ->
    ctx.clearRect(0, 0, 300, 300)
    ctx.beginPath()
    ctx.arc(x, y, 10, 0, Math.PI*2, true)
    ctx.closePath()
    ctx.fill()

    x += dx;
    y += dy;

window.onload = ->
    init()
