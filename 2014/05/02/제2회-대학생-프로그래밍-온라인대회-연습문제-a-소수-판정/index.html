문제

소스 코드

#include <iostream>
#include <cmath>
using namespace std;

bool IsPrime(int);

int main() {

    int T;// 테스트 케이스 개수(1 <= T <= 10)
    cin >> T;

    for(int i = 0; i < T; i++)
    {
        int N;// 소수인지 판정할 자연수(1 <= N <= 100,000,000)
        cin >> N;

        if(IsPrime(N))
        {
            cout << "YES" << endl;
        }
        else
        {
            cout << "NO" << endl;
        }
    }

    return 0;
}

bool IsPrime(int n)
{
    for(int i = 2; i < sqrt(n); i++)
    {
        if(n % i == 0)
        {
            return false;
        }
    }
    return true;
}

코멘트

소수를 구하는 문제이다. 이 문제의 키 포인트는 제곱근 이상은 해볼 필요가 없다는 사실인것 같다.
